#1936/01/01
country_event = {
	id = exact_date_trigger.1
	hidden = yes
	fire_only_once = yes
	
	trigger = {
		date > 1936.1.1
	}
	
	option = {
		if = {
			limit = { 
				country_exists = MON
			}
			MON = { country_event = mongolia.1 }
		}
		
		if = {
			limit = { 
				country_exists = SOV
			}
			SOV = { country_event = muscovy.1 }
		}
		
		if = {
			limit = { 
				country_exists = AST
			}
			AST = { country_event = ostcasa.1 }
		}
		
	}
}

#1936/04/04
country_event = {
	id = exact_date_trigger.95
	hidden = yes
	fire_only_once = yes
	
	trigger = {
		date > 1936.4.3
	}
	
	option = {
		if = {
			limit = { 
				country_exists = AST
				OR = {
					NOT = {
						OR = {
							has_country_flag = AST_federalistoian_wins_8th
							has_country_flag = AST_centralistian_wins_8th
							has_country_flag = AST_labouritian_wins_8th
						}
					}
					
					has_country_flag = AST_committee_elections_deadlock
				}
			}
			AST = { country_event = ostcasa.9 }
		}
		
		if = {
			limit = { 
				country_exists = AST
				has_country_flag = AST_federalistoian_wins_8th
			}
			AST = { country_event = ostcasa.18 }
		}
		
		if = {
			limit = { 
				country_exists = AST
				has_country_flag = AST_centralistian_wins_8th
			}
			AST = { country_event = ostcasa.19 }
		}
		
		if = {
			limit = { 
				country_exists = AST
				has_country_flag = AST_labouritian_wins_8th
			}
			AST = { country_event = ostcasa.20 }
		}
	}
}
