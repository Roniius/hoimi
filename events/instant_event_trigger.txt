#1936/01/01
country_event = {
	id = exact_date_trigger.1
	hidden = yes
	fire_only_once = yes
	
	trigger = {
		date > 1936.1.1
	}
	
	option = {
		if = {
			limit = { 
				country_exists = MON
			}
			MON = { country_event = mongolia.1 }
		}
		
		if = {
			limit = { 
				country_exists = SOV
			}
			SOV = { country_event = muscovy.1 }
		}
		
	}
}
